CREATE TABLE AUTHORS(
AUTHOR_ID      NUMBER(5),
AUTHOR_LNAME   VARCHAR2(20) NOT NULL,
AUTHOR_FNAME   VARCHAR(20),
GENDER         CHAR(1) DEFAULT 'F',
DOB            DATE,            -- date of birth
DOD            DATE,             -- date of death
COUNTRY        VARCHAR2(20) DEFAULT 'GREECE'
);

CREATE TABLE BOOKS(
ISBN         NUMBER(5),        -- id for books
TITLE        VARCHAR2(50),
AUTHOR_ID    NUMBER(5),
GENRE        VARCHAR2(40),     -- type of book
DOP          DATE,           -- date of published
EDITION      NUMBER(2),
PRICE        NUMBER(6,2),
PAGES        NUMBER(5) 
);

CREATE TABLE COPY(
COPY_ID        VARCHAR2(9),    --ISBN(5)+_+001
ISBN           NUMBER(5)
);

CREATE TABLE USERS(
USER_ID         NUMBER(10),
USER_LNAME      VARCHAR2(20) NOT NULL,
USER_FNAME      VARCHAR2(20),
GENDER          CHAR(1),
CONTACT_NO      VARCHAR2(15),
EMAIL           VARCHAR2(25),
DOB             DATE
);

CREATE TABLE LOANS(
USER_ID      NUMBER(10),  
LOAN_ID      NUMBER(10),
COPY_ID      VARCHAR2(9),
ISBN         NUMBER(5),
DOL          DATE NOT NULL,    -- date of loan
DOR          DATE     -- date of return
);



-- SEQUENCES    

CREATE SEQUENCE SEQ_USER_ID
INCREMENT BY 1
START WITH 10000
NOCYCLE;

CREATE SEQUENCE SEQ_AUTHOR_ID
INCREMENT BY 1
START WITH 20000
NOCYCLE;

CREATE SEQUENCE SEQ_LOAN_ID
INCREMENT BY 1
START WITH 30000
NOCYCLE;

--DESCRIBE

DESC AUTHORS;
DESC BOOKS;
DESC COPY;
DESC USERS;
DESC LOANS;
